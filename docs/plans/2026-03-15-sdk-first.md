# Clawforce SDK Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a clean public SDK for Clawforce that wraps existing internals with abstract vocabulary, in-process event subscriptions, and proper TypeScript types — making Clawforce usable as a library without HTTP or the dashboard.

**Architecture:** A `Clawforce` class provides domain-scoped access to namespaced APIs (tasks, events, budget, trust, agents, goals, messages, config, monitoring). Each namespace wraps existing internal functions, pre-filling projectId and handling errors. The dashboard becomes an optional plugin that consumes the SDK. Vocabulary is abstract from day one (group/role/coordinator, not department/extends/manager).

**Tech Stack:** TypeScript, SQLite (via existing `getDb`), Vitest

---

```

  index.ts              # Clawforce class, init(), namespace wiring
  types.ts              # Public SDK types (abstract vocabulary)
  tasks.ts              # cf.tasks namespace
  events.ts             # cf.events namespace (with .on() subscriptions)
  budget.ts             # cf.budget namespace
  agents.ts             # cf.agents namespace
  trust.ts              # cf.trust namespace
  goals.ts              # cf.goals namespace
  messages.ts           # cf.messages namespace
  config.ts             # cf.config namespace
  monitoring.ts         # cf.monitoring namespace
  capabilities.ts       # Capability system (hasCapability, getCapabilities)
```

Each namespace file exports a class that takes `(projectId, getDb)` in constructor and wraps internal functions.

```typescript
// src/sdk/index.ts
import type { ClawforceOptions } from "./types.js";
import { TasksNamespace } from "./tasks.js";
import { EventsNamespace } from "./events.js";
import { BudgetNamespace } from "./budget.js";
import { AgentsNamespace } from "./agents.js";
import { TrustNamespace } from "./trust.js";
import { GoalsNamespace } from "./goals.js";
import { MessagesNamespace } from "./messages.js";
import { MonitoringNamespace } from "./monitoring.js";

export class Clawforce {
  readonly domain: string;
  private readonly opts: ClawforceOptions;

  // Lazy-initialized namespaces
  private _tasks?: TasksNamespace;
  private _events?: EventsNamespace;
  private _budget?: BudgetNamespace;
  private _agents?: AgentsNamespace;
  private _trust?: TrustNamespace;
  private _goals?: GoalsNamespace;
  private _messages?: MessagesNamespace;
  private _monitoring?: MonitoringNamespace;

  private constructor(opts: ClawforceOptions) {
    this.opts = opts;
    this.domain = opts.domain;
  }

  static init(opts: ClawforceOptions): Clawforce {
    return new Clawforce(opts);
  }

  get tasks(): TasksNamespace {
    return (this._tasks ??= new TasksNamespace(this.domain));
  }

  get events(): EventsNamespace {
    return (this._events ??= new EventsNamespace(this.domain));
  }

  get budget(): BudgetNamespace {
    return (this._budget ??= new BudgetNamespace(this.domain));
  }

  get agents(): AgentsNamespace {
    return (this._agents ??= new AgentsNamespace(this.domain));
  }

  get trust(): TrustNamespace {
    return (this._trust ??= new TrustNamespace(this.domain));
  }

  get goals(): GoalsNamespace {
    return (this._goals ??= new GoalsNamespace(this.domain));
  }

  get messages(): MessagesNamespace {
    return (this._messages ??= new MessagesNamespace(this.domain));
  }

  get monitoring(): MonitoringNamespace {
    return (this._monitoring ??= new MonitoringNamespace(this.domain));
  }
}

// Re-export types
export * from "./types.js";
```

```typescript
// src/sdk/tasks.ts
import type { Task, TaskParams, TaskState } from "./types.js";
import {
  createTask, getTask, listTasks, transitionTask, reassignTask,
  getTaskEvidence, getTaskTransitions,
} from "../tasks/ops.js";

// Vocabulary mapping: abstract → internal
function toInternalParams(domain: string, params: TaskParams, actor?: string) {
  return {
    projectId: domain,
    title: params.title,
    description: params.description,
    priority: params.priority,
    assignedTo: params.assignedTo,
    department: params.group,
    team: params.subgroup,
    goalId: params.goalId,
    tags: params.tags,
    deadline: params.deadline,
    metadata: params.metadata,
    createdBy: actor ?? "sdk",
  };
}

function toPublicTask(internal: Record<string, unknown>): Task {
  return {
    id: internal.id as string,
    title: internal.title as string,
    state: internal.state as TaskState,
    priority: internal.priority as string,
    assignedTo: internal.assigned_to as string | undefined,
    group: internal.department as string | undefined,
    subgroup: internal.team as string | undefined,
    goalId: internal.goal_id as string | undefined,
    tags: (internal.tags as string[]) ?? [],
    createdAt: internal.created_at as number,
    updatedAt: internal.updated_at as number,
    metadata: internal.metadata as Record<string, unknown> | undefined,
  };
}

export class TasksNamespace {
  constructor(readonly domain: string) {}

  create(params: TaskParams, actor?: string): Task {
    const result = createTask(toInternalParams(this.domain, params, actor));
    return toPublicTask(result as Record<string, unknown>);
  }

  get(taskId: string): Task | undefined {
    const result = getTask(this.domain, taskId);
    return result ? toPublicTask(result as Record<string, unknown>) : undefined;
  }

  list(filters?: { state?: TaskState; assignedTo?: string; group?: string; limit?: number }): Task[] {
    const internal = listTasks(this.domain, {
      state: filters?.state,
      assignee: filters?.assignedTo,
      department: filters?.group,
      limit: filters?.limit,
    });
    return internal.map((t) => toPublicTask(t as Record<string, unknown>));
  }

  transition(taskId: string, toState: TaskState, opts?: { actor?: string; reason?: string }): Task {
    const result = transitionTask({
      projectId: this.domain,
      taskId,
      toState,
      actor: opts?.actor ?? "sdk",
      reason: opts?.reason,
    });
    if (!result.ok) throw new Error(result.reason ?? "Transition failed");
    return this.get(taskId)!;
  }

  reassign(taskId: string, newAssignee: string, opts?: { actor?: string; reason?: string }): Task {
    const result = reassignTask({
      projectId: this.domain,
      taskId,
      newAssignee,
      actor: opts?.actor ?? "sdk",
      reason: opts?.reason,
    });
    if (!result.ok) throw new Error(result.reason ?? "Reassign failed");
    return this.get(taskId)!;
  }

  evidence(taskId: string) { return getTaskEvidence(this.domain, taskId); }
  history(taskId: string) { return getTaskTransitions(this.domain, taskId); }
}
```

```typescript
// src/sdk/events.ts
import type { ClawforceEvent, EventHandler, EventParams } from "./types.js";
import { ingestEvent, listEvents, countEvents } from "../events/store.js";

export class EventsNamespace {
  private listeners = new Map<string, Set<EventHandler>>();

  constructor(readonly domain: string) {}

  emit(type: string, payload?: Record<string, unknown>, opts?: { source?: string; dedupKey?: string }): ClawforceEvent {
    const result = ingestEvent(
      this.domain,
      type,
      opts?.source ?? "sdk",
      payload ?? {},
      opts?.dedupKey,
    );

    // Build event object for subscribers
    const event: ClawforceEvent = {
      id: result.id,
      type,
      source: opts?.source ?? "sdk",
      payload: payload ?? {},
      status: result.deduplicated ? "deduplicated" : "pending",
      createdAt: Date.now(),
    };

    // Fire subscribers (type-specific + wildcard)
    this.fireListeners(type, event);
    if (type !== "*") this.fireListeners("*", event);

    return event;
  }

  list(filters?: { type?: string; status?: string; limit?: number; offset?: number }): ClawforceEvent[] {
    return listEvents(this.domain, {
      type: filters?.type,
      status: filters?.status,
      limit: filters?.limit ?? 50,
      offset: filters?.offset,
    }) as ClawforceEvent[];
  }

  count(filters?: { type?: string; status?: string }): number {
    return countEvents(this.domain, filters);
  }

  on(type: string, handler: EventHandler): void {
    if (!this.listeners.has(type)) this.listeners.set(type, new Set());
    this.listeners.get(type)!.add(handler);
  }

  off(type: string, handler: EventHandler): void {
    this.listeners.get(type)?.delete(handler);
  }

  private fireListeners(type: string, event: ClawforceEvent): void {
    const handlers = this.listeners.get(type);
    if (handlers) {
      for (const handler of handlers) {
        try { handler(event); } catch { /* subscriber errors don't propagate */ }
      }
    }
  }
}
```
